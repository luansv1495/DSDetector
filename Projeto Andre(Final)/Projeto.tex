\section{Objetivo}

\frame {
\frametitle{Objetivo}

	\begin{itemize}
				\item O objetivo deste trabalho é elaborar um sistema embarcado para detecção de concentração de álcool baseado no Arduino, que futuramemte possa ser implantado em um veículo para que apenas se locomova caso o motorista não esteja sobre efeito de alcool, assim evitando um possível acidente.
\end{itemize}	
}

\section{Fundamentação teórica}
\subsection{Lei Nº 11.705}
\frame {
\frametitle{Lei Nº 11.705 (Lei Seca)}
	\begin{itemize}
	\item Está lei altera a Lei no 9.503, de 23 de setembro de 1997, e a Lei no 9.294, de 15 de julho de 1996, nos termos do § 4º do art. 220 da Constituição Federal, para inibir o consumo de bebida alcoólica por condutor de veículo automotor, e dá outras providências. 
	\end{itemize}	
}

\frametitle{Punição}
	\begin{itemize}
	\item De acordo com a nova legislação, o motorista que for flagrado com nível de álcool acima do permitido (0,1 mg/l de sangue) terá que pagar uma multa de R\$ 2.934,70 eterá a carteira de habilitação suspensa pelo prazo de 12 meses.
	\item O pior é para quem estiver embriagado (níveis acima de 0,3 mg/l): o motorista corre o risco de ser preso, e a detenção é de 6 meses a 1 ano. 
	\end{itemize}

\section{Drive Safe Detector(DSDetector)}
\frame {
\frametitle{Drive Safe Detector}
	 \begin{itemize}
		\item O DSDetector, um sistema embarcado projetado em c++, utiliza o paradigma de programação orientado a objetos, feito totalmente na plataforma Arduino. Ele é composto por dois periféricos de entrada, Sensor de gás MQ-3 e sensor de Temperatura, três periféricos de saída, Display Lcd, Buzzer Beep e Led Vermelho, e demais componentes.
\end{itemize}	
}

\section{Resultados}
\frame {
\frametitle{Metodologia dos Testes}
	 \begin{itemize}
		\item Foram feitas coletas de dados de acordo com os seguintes intervalos de temperatura, temperatura menor ou igual a 19ºC, maior que 19ºC e menor que 27ºC e maior ou igual a 27ºC, intervalos os quais foram verificados mudanças na coleta do sensor MQ-3. 
\end{itemize}	
}
\note{
}
\frame {
\frametitle{Fórmula}
	 \begin{itemize}
		\item BAC(em mg/L)=BrAc(em mg/L)*FATOR. 
		\item BAC = é a concentração de álcool no sangue.
		\item BrAc = é a concentração de álcool na
respiração.
		\item FATOR = varia de país para país no caso do Brasil é 0,3.
\end{itemize}	
}
